70 inline float Vec3f:: getPlaneDistance(Vec3f Normal)//clase vector
71 {
72  float distance = 0; // This variable holds the distance from the plane tot he origin
73 
74  // Use the plane equation to find the distance (Ax + By + Cz + D = 0) We want to find D.
75  // So, we come up with D = -(Ax + By + Cz)
76  // Basically, the negated dot product of the normal of the plane and the point. (More about the dot product in another tutorial)
77  distance = -((Normal.x * x) + (Normal.y * y) + (Normal.z * z));
78 
79  return distance; // Return the distance
80 }

88 inline Vec3f Vec3f::getClosestPointOnLine(Vec3f vA, Vec3f vB)
89 {
90  // Create the vector from end point vA to our point vPoint.
91  Vec3f vVector1 = *this - vA;
92 
93  // Create a normalized direction vector from end point vA to end point vB
94  Vec3f vVector2 = (vB - vA).normalize();
95 
96 
97  // Use the distance formula to find the distance of the line segment (or magnitude)
98  float d = vA.distance(vB);
99 
100 
101  // Using the dot product, we project the vVector1 onto the vector vVector2.
102  // This essentially gives us the distance from our projected vector from vA.
103  float t =vVector2.dot(vVector1);
104 
105  // If our projected distance from vA, "t", is less than or equal to 0, it must
106  // be closest to the end point vA. We want to return this end point.
107  if (t <= 0)
108  return vA;
109 
110  // If our projected distance from vA, "t", is greater than or equal to the magnitude
111  // or distance of the line segment, it must be closest to the end point vB. So, return vB.
112  if (t >= d)
113  return vB;
114 
115  // Here we create a vector that is of length t and in the direction of vVector2
116  Vec3f vVector3 = vVector2 * t;
117 
118  // To find the closest point on the line segment, we just add vVector3 to the original
119  // end point vA.
120  Vec3f vClosestPoint = vA + vVector3;
121 
122  // Return the closest point on the line segment
123  return vClosestPoint;
124 }

128 inline double Vec3f::getAngleBetweenVectors(Vec3f vector)
129 {
130  // Get the dot product of the vectors
131  double dotProduct = dot(vector);
132 
133  // Get the product of both of the vectors magnitudes
134  double vectorsMagnitude = length() * vector.length() ;
135 
136  // Get the angle in radians between the 2 vectors
137  double angle = acos( dotProduct / vectorsMagnitude );
138 
139  // Here we make sure that the angle is not a -1.#IND0000000 number, which means indefinate
140  if(isnan(angle))
141  return 0;
142 
143  // Return the angle in radians
144  return( angle );
145 }

151 inline float invSqrt(float x)
152 {
153  float xhalf = 0.5f * x;
154  int i = *(int*)&x; // get bits for floating value
155  i = 0x5f3759df - (i>>1); // gives initial guess

157  x = x * (1.5f - xhalf*x*x); // Newton step
