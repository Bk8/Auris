43  bool intersectsTriangle(Ray* ray, double & dist,Vec3f& intersection){
44  /*MöllerTrumbore intersection algorithm
45  thanks wikipedia :)
46  https://en.wikipedia.org/wiki/M%C3%B6ller%E2%80%93Trumbore_intersection_algorithm
47  */
48  double e = 0.00001;
49 
50  Vec3f v1 = vertexes[0];
51  Vec3f v2 = vertexes[1];
52  Vec3f v3 = vertexes[2];
53 
54  Vec3f e1 = v2- v1;
55  Vec3f e2 = v3- v1;
56 
57  //Begin calculating determinant - also used to calculate u parameter
58  Vec3f p = ray->direction.cross(e2);
59  double det = e1.dot(p);
60 
61  //If determinant is near zero, ray lies in plane of triangle
62  if (det > -e && det < e)
63  return false;
64  double inv_det = 1.0 / det;
65 
66  Vec3f t = ray->startPos-v1;
67  double u = t.dot(p) * inv_det;
68  //The intersection lies outside of the triangle
69  if (u < 0.0 || u > 1.0)
70  return false;
71 
72  Vec3f q = t.cross(e1);
73  double v = ray->direction.dot(q) * inv_det;
74  //The intersection lies outside of the triangle
75  if (v < 0.0 || u + v > 1.0)
76  return false;
77 
78  double a = e2.dot(q) * inv_det;
79 
80  //http://gamedev.stackexchange.com/questions/114955/m%C3%B6ller-trumbore-intersection-point
81 
82  intersection = v1+u*e2+v*e1;
83 
84  if (a > e){
85  dist = a;
86  return true;
87  }
88 
89  return false;
90  }
